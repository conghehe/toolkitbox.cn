常见问题


Q: 生成的代码可以直接使用吗？
A: 可以。生成的HTML代码包含完整的表单结构、CSS样式和HTML5验证属性，粘贴到网页即可使用。如需后端处理，只需添加<form action="...">属性和服务器端处理脚本。


Q: 数据安全吗？
A: 完全安全。所有操作均在浏览器本地完成，不会上传任何数据到服务器。表单预览也只是前端的CSS/HTML展示。


Q: 支持移动端吗？
A: 支持。生成的表单使用响应式设计，在手机和平板上自动适配布局。本工具本身也针对移动端做了优化。

// ============ CLIPBOARD ============
function copyToClipboard(){
  if(fields.length === 0){
    showToast('⚠️ 请先添加至少一个字段');
    return;
  }
  // Generate code without showing modal
  let html = '\n\n\n\n\n表单\n\n\n\n
表单
\n
\n';
  fields.forEach(f => {
    html += '  
\n    \n';
    const req = f.required ? ' required' : '';
    const ph = f.placeholder ? ' placeholder="'+escAttr(f.placeholder)+'"' : '';
    if(f.type === 'textarea') html += '    \n';
    else if(f.type === 'select'){
      html += '    \n';
    }else if(f.type === 'checkbox'){
      html += '    
\n';
      f.options.forEach(o => { html += '      \n'; });
      html += '    
\n';
    }else if(f.type === 'radio'){
      html += '    
\n';
      f.options.forEach(o => { html += '      \n'; });
      html += '    
\n';
    }else{
      html += '    \n';
    }
    html += '  
\n';
  });
  html += '  \n
\n\n';
  navigator.clipboard.writeText(html).then(()=>showToast('✅ 已复制 '+fields.length+' 个字段的HTML代码！'));
}

function esc(s){ if(!s) return ''; return s.replace(/&/g,'&').replace(//g,'>'); }
function escAttr(s){ if(!s) return ''; return s.replace(/&/g,'&').replace(/"/g,'"').replace(//g,'>'); }
function updateCounts(){

function showToast(msg){
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2000);
}
function toggleTheme(){
  isDark = !isDark;
  document.documentElement.classList.toggle('dark', isDark);
  document.getElementById('themeBtn').textContent = isDark ? '☀️' : '🌙';
}
function showMobileTab(tab){
  document.querySelectorAll('.mobile-tab').forEach(b => b.classList.remove('active'));
  event.target.classList.add('active');
  const lp = document.getElementById('leftPanel');
  const cp = document.getElementById('centerPanel');
  const rp = document.getElementById('rightPanel');
  lp.style.display = tab === 'fields' ? 'flex' : 'none';
  cp.style.display = tab === 'builder' ? 'flex' : 'none';
  rp.style.display = tab === 'props' ? 'flex' : 'none';
  rp.classList.remove('mobile-show');
}
